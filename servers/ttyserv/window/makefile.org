NONE		= 0x0
TCAP		= 0x1
TCAP_ALL	= 0x2
INPUT		= 0x4
INPUT_ALL	= 0x8
TERM		= 0x10
TERM_ALL	= 0x20
ANSI		= 0x40
ANSI_ALL	= 0x80
READ		= 0x100
READ_ALL	= 0x200
WRITE		= 0x400

OPEN		= 0x1000
CREATE		= 0x4000
DELETE		= 0x10000
ATTR		= 0x40000
SETTERM		= 0x100000
CBREAK		= 0x400000
MAIN		= 0x1000000
MAIN_ALL	= 0x2000000
ERROR		= 0x80000000

#DLEVEL		= DEBUG=0xFFFFFFFF
#DLEVEL		= DEBUG=$(MAIN_ALL)

#+$(ANSI)+$(WRITE)+$(DOPRIVATE)+$(READ)

DLEVEL		= DEBUG=$(NONE)

DFLAGS		= -dSYSDEBUG
NCFLAGS 	= -t4 -ps1 -pg0 -j$(INCLUDE)/ -d$(DLEVEL) $(DFLAGS)
LINKOPTS	= -v -f -s 4000

#default rules

.c.x:
	$(CC) $(NCFLAGS) $*.c -s$*.x >>& $(ERRORS)

.c.o:  
	$(CC) $(NCFLAGS) $*.c -s$(FIFO)/$*.p >>& $(ERRORS) && \
	$(ASM) -p $(FIFO)/$*.p -o$@ && \
	$(RM) $(FIFO)/$*.p

.a.o:
	$(AMPP) -i$(KINCLUDE)/ $(KINCLUDE)/basic.m $*.a | \
	$(ASM) -p -o $@

.a.x:
	$(AMPP) -o$@ -i$(KINCLUDE)/ $(KINCLUDE)/basic.m $*.a

.x.o:
	$(ASM) -p $*.x -o$@

.o.z:
	asm $(LINKOPTS) $(CSTART) $*.o -o$@


default: delerror iwindow.i copywin setterm diag_wm showerror

$(OFILES)	: $(HFILES)

delerror	:
	-$(RM) -f $(ERRORS)

showerror:
	-grep '[0-9a-z_/][0-9a-z_/]\.[cho]' $(ERRORS)

iwindow.i	: $(OFILES)
	asm $(LINKOPTS) $(SSTART) $(OFILES) -o iwindow.i

dwindow.i	: $(OFILES)
	asm $(LINKOPTS) $(SSTART) $(OFILES) -o dwindow.i

	$(RM) $(FIFO)/$*.p
